                    <!-- jQuery -->
        <script src="./js/jquery.js"></script>

                    <!-- ma5menu -->
        <link href="./css/ma5-menu.min.css" rel="stylesheet" type="text/css">

                    <!-- Call the script -->
        <script>
            $(document).ready(function () {

            });
